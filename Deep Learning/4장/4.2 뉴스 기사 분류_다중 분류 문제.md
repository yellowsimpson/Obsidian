```python
def to_one_hot(labels, dimension=46):
	results = np.zeros((len(labels), dimension))
	for i, label in enumerate(labels):
		results[i, label] = 1.
	return results
y_train = to_one_hot(train_labels)
y_test = to_one_hot(test_labels)
```
